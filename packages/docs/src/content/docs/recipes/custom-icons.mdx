### Loading the icon

We load it locally and transform it.

```ts title="uno.config.ts"
import fs from 'node:fs/promises'
import { defineConfig, presetIcons } from 'unocss'
import { presetStarlightIcons } from 'starlight-plugin-icons/uno'

export default defineConfig({
  presets: [
    presetStarlightIcons(),
    presetIcons({
      collections: {
        icons: {
          drizzle: () => fs.readFile('./src/assets/drizzle.svg', 'utf-8'),
        },
      },
      extraProperties: { display: 'inline-block', 'vertical-align': 'middle' },
      customizations: {
        iconCustomizer(collection, icon, props) {
          if (collection === 'icons' && icon === 'drizzle')
            props.transform = 'scale(0.8)'
        },
      },
    }),
  ],
})
```

<br />

<Aside type="tip">
  These collections are also recommended to scale down if you are using them:

  ```ts
  if (['devicon', 'simple-icons', 'logos'].includes(collection))
    props.transform = 'scale(0.8)'
  ```
</Aside>

### Customizing with CSS

Now, we need to color it correctly depending on the theme and whether the link is active or not.

```css title="src/styles/custom.css"
.i-icons\:drizzle {
  color: #c5f74f !important;
}

html[data-theme='light'] .i-icons\:drizzle {
  color: var(--sl-color-white) !important;
}

a[aria-current='page'] > .i-icons\:drizzle {
  color: var(--sl-color-text-invert) !important;
}
```
