This guide will walk you through installing `starlight-plugin-icons` and setting it up for the first time.

<br />

<Steps>

1. ##### Install the package

    <Tabs syncKey="pkg">
      <TabItem label="npm">
        ```sh
        npm i starlight-plugin-icons
        ```
      </TabItem>
      <TabItem label="pnpm">
        ```sh
        pnpm add starlight-plugin-icons
        ```
      </TabItem>
      <TabItem label="yarn">
        ```sh
        yarn add starlight-plugin-icons
        ```
      </TabItem>
      <TabItem label="bun">
        ```sh
        bun add starlight-plugin-icons
        ```
      </TabItem>
      <TabItem label="ni">
        ```sh
        ni starlight-plugin-icons
        ```
      </TabItem>
    </Tabs>

2. ##### Codeblocks

    ```js title="astro.config.mjs"
    import { defineConfig } from 'astro/config'
    import starlight from '@astrojs/starlight'

    export default defineConfig({
      integrations: [starlight()],
    })
    ```
